# 큰 수의 법칙

#문제: 배열의 크N, 숫자가 더해지는 횟수M, 그리고 K가 주어질 큰 수의 법칙에 따른 결과를 출력하시오

#입력조건 : 책 참조

#문제아이디어:  가장 큰 수를 K번 더하고, 두번째로 큰 수를 한번 더 하는 연산

n, m, k = map(int, input().split())

data = list(map(int, input().split()))

         
data.sort()
            
first = data[n - 1] # 가장 큰 수
second = data[n - 2] # 두 번째로 큰 수
            
result = 0
            
while True:
    for i in range(k): # 가장 큰 수를 k번 더하기
            if m == 0:
                break
            result += first
            m -= 1 #더할 때마다 1씩 빼기
    if m == 0:
            break
    result += second # 두 번째로 큰 수를 한 번 더하기.
    m -= 1

print(result)  # 최종 답안 출력
